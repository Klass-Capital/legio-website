<script>

  function toggleDropdown() { document.getElementById('moduleDropdown').classList.toggle('open'); }

  document.addEventListener('click', function(e) {

    const d = document.getElementById('moduleDropdown');

    if (!d.contains(e.target)) d.classList.remove('open');

  });


  const obs = new IntersectionObserver(entries => {

    entries.forEach(e => { if (e.isIntersecting) e.target.classList.add('visible'); });

  }, { threshold: 0.1 });

  document.querySelectorAll('.fade-in').forEach(el => obs.observe(el));


  function animateCounter(el) {

    const target = +el.dataset.target, dur = 1400, start = performance.now();

    function tick(now) {

      const p = Math.min((now - start) / dur, 1);

      const ease = 1 - Math.pow(1 - p, 3);

      el.textContent = Math.floor(ease * target);

      if (p < 1) requestAnimationFrame(tick); else el.textContent = target;

    }

    requestAnimationFrame(tick);

  }

  const statsObs = new IntersectionObserver(entries => {

    entries.forEach(e => {

      if (e.isIntersecting) {

        e.target.querySelectorAll('.counter').forEach(animateCounter);

        statsObs.unobserve(e.target);

      }

    });

  }, { threshold: 0.3 });

  document.querySelectorAll('.stats-grid').forEach(el => statsObs.observe(el));

</script>
