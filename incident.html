var obs = new IntersectionObserver(function(entries) {

  entries.forEach(function(e) { if (e.isIntersecting) e.target.classList.add('visible'); });

}, { threshold: 0.1 });

document.querySelectorAll('.fade-in').forEach(function(el) { obs.observe(el); });


window.onload = function() {

  var cv = document.getElementById('linkCanvas');

  var cx = cv.getContext('2d');

  var alertEl = document.getElementById('patternAlert');

  var statusDot = document.getElementById('statusDot');

  var statusText = document.getElementById('statusText');

  var W = 800, H = 460;


  function pill(x,y,w,h,r) {

    cx.beginPath();

    cx.moveTo(x+r,y); cx.lineTo(x+w-r,y);

    cx.arcTo(x+w,y,x+w,y+r,r); cx.lineTo(x+w,y+h-r);

    cx.arcTo(x+w,y+h,x+w-r,y+h,r); cx.lineTo(x+r,y+h);

    cx.arcTo(x,y+h,x,y+h-r,r); cx.lineTo(x,y+r);

    cx.arcTo(x,y,x+r,y,r); cx.closePath();

  }


  var N = [

    { id:0,  label:'Pattern',     sub:'Evening / no lock cut',  r:28, col:'#5b8aff', c:true,  x:400, y:220 },

    { id:1,  label:'Bike Theft',  sub:'Lot C — Tue 9:12pm',    r:18, col:'#7ba3ff', c:true,  x:170, y:100 },

    { id:5,  label:'Bike Theft',  sub:'Library — Mon 9:05pm',  r:15, col:'#7ba3ff', c:true,  x:400, y:60  },

    { id:6,  label:'Bike Theft',  sub:'Union — Wed 10:10pm',   r:14, col:'#7ba3ff', c:true,  x:400, y:388 },

    { id:7,  label:'Vandalism',   sub:'Gym',                   r:11, col:'#44446a', c:false, x:750, y:230 },

    { id:8,  label:'Trespass',    sub:'Dorm B',                r:10, col:'#44446a', c:false, x:55,  y:225 },

    { id:9,  label:'Assault',     sub:'Quad',                  r:10, col:'#44446a', c:false, x:700, y:400 },

    { id:10, label:'Drug',        sub:'Parking A',             r:10, col:'#44446a', c:false, x:100, y:420 },

    { id:11, label:'Trespass',    sub:'Science Hall',          r:9,  col:'#44446a', c:false, x:720, y:60  },

    { id:12, label:'Vandalism',   sub:'Athletic Ctr',          r:9,  col:'#44446a', c:false, x:80,  y:65  },

  ];


  var E = [

    { a:1,  b:0, conf:0.93, s:true  },

    { a:2,  b:0, conf:0.91, s:true  },

    { a:3,  b:0, conf:0.88, s:true  },

    { a:4,  b:0, conf:0.84, s:true  },

    { a:5,  b:0, conf:0.79, s:true  },

    { a:6,  b:0, conf:0.76, s:true  },

    { a:7,  b:9, conf:0.31, s:false },

    { a:8,  b:10,conf:0.24, s:false },

    { a:11, b:7, conf:0.19, s:false },

    { a:12, b:8, conf:0.17, s:false },

  ];


  var state='appear', tick=0, shown=false;

  var na=[], ep=[], ba=[];

  for(var i=0;i<N.length;i++) na.push(0);

  for(var i=0;i<E.length;i++){ep.push(0);ba.push(0);}


  function drawBg() {

    cx.fillStyle='#07090f'; cx.fillRect(0,0,W,H);

    var g=cx.createRadialGradient(400,220,0,400,220,280);

    g.addColorStop(0,'rgba(59,124,255,0.06)');

    g.addColorStop(1,'rgba(0,0,0,0)');

    cx.fillStyle=g; cx.fillRect(0,0,W,H);

  }


  function drawEdge(e,prog,alpha) {

    if(!alpha||!prog) return;

    var a=N[e.a],b=N[e.b];

    var ex=a.x+(b.x-a.x)*prog, ey=a.y+(b.y-a.y)*prog;

    cx.save();

    cx.globalAlpha=alpha*(e.s?0.45:0.1);

    cx.beginPath(); cx.moveTo(a.x,a.y); cx.lineTo(ex,ey);

    cx.strokeStyle=e.s?'#5b8aff':'rgba(160,160,200,0.8)';

    cx.lineWidth=e.s?1.5:0.8;

    if(!e.s) cx.setLineDash([4,6]);

    cx.stroke(); cx.setLineDash([]);

    cx.restore();

  }


  function drawBadge(e,alpha) {

    if(!alpha||!e.s) return;

    var a=N[e.a],b=N[e.b];

    var mx=(a.x+b.x)/2, my=(a.y+b.y)/2;

    var txt=Math.round(e.conf*100)+'%';

    cx.save(); cx.globalAlpha=alpha;

    cx.font='600 10px sans-serif';

    var tw=cx.measureText(txt).width;

    cx.fillStyle='#07090f';

    pill(mx-tw/2-7,my-9,tw+14,18,4); cx.fill();

    cx.strokeStyle='rgba(91,138,255,0.4)'; cx.lineWidth=1;

    pill(mx-tw/2-7,my-9,tw+14,18,4); cx.stroke();

    cx.fillStyle='#93b4ff'; cx.textAlign='center'; cx.textBaseline='middle';

    cx.fillText(txt,mx,my);

    cx.restore();

  }


  function drawNode(n,alpha,glow) {

    if(!alpha) return;

    cx.save(); cx.globalAlpha=alpha;

    var glr=n.r*(n.c?4:2.5);

    var g=cx.createRadialGradient(n.x,n.y,n.r*0.2,n.x,n.y,glr);

    g.addColorStop(0,n.col+(n.c?'28':'14'));

    g.addColorStop(1,'rgba(0,0,0,0)');

    cx.globalAlpha=alpha*(n.c?0.7+glow*0.3:0.3);

    cx.beginPath(); cx.arc(n.x,n.y,glr,0,Math.PI*2);

    cx.fillStyle=g; cx.fill();

    cx.globalAlpha=alpha;

    cx.beginPath(); cx.arc(n.x,n.y,n.r,0,Math.PI*2);

    cx.fillStyle=n.col+(n.c?'1e':'12'); cx.fill();

    cx.strokeStyle=n.col+(n.c?'bb':'44'); cx.lineWidth=n.c?1.5:1; cx.stroke();

    if(n.id===0){

      cx.beginPath(); cx.arc(n.x,n.y,n.r*0.38,0,Math.PI*2);

      cx.fillStyle=n.col+'99'; cx.fill();

    }

    if(n.c){

      cx.fillStyle=n.id===0?'#ffffff':'#c0d0ff';

      cx.font=(n.id===0?'600 11px':'500 10px')+' sans-serif';

      cx.textAlign='center'; cx.textBaseline='middle';

      cx.fillText(n.label,n.x,n.y-4);

      cx.fillStyle='rgba(255,255,255,0.36)';

      cx.font='9px sans-serif';

      cx.fillText(n.sub.split('—')[0].trim(),n.x,n.y+7);

    }

    cx.restore();

  }


  function frame() {

    requestAnimationFrame(frame);

    cx.clearRect(0,0,W,H);

    drawBg();

    tick++;

    var glow=(Math.sin(tick*0.04)+1)/2;


    if(state==='appear'){

      for(var i=0;i<N.length;i++) na[i]=Math.min(1,Math.max(0,(tick-i*7)/22));

      if(na[N.length-1]>=1&&tick>90){state='trace';tick=0;if(statusText)statusText.textContent='Analyzing patterns';}

      for(var i=0;i<N.length;i++) drawNode(N[i],na[i],0);


    } else if(state==='trace'){

      for(var i=0;i<E.length;i++){

        var gate=i===0?1:ep[i-1];

        if(gate>=0.5) ep[i]=Math.min(1,ep[i]+0.022);

        if(ep[i]>=1) ba[i]=Math.min(1,ba[i]+0.04);

      }

      for(var i=0;i<E.length;i++) drawEdge(E[i],ep[i],1);

      for(var i=0;i<E.length;i++) drawBadge(E[i],ba[i]);

      for(var i=0;i<N.length;i++) drawNode(N[i],1,glow);

      if(ep[E.length-1]>=1&&tick>80){

        state='highlight';tick=0;

        if(statusText)statusText.textContent='Pattern detected';

        if(statusDot)statusDot.classList.add('green');

        if(!shown&&alertEl){shown=true;alertEl.classList.add('visible');}

      }


    } else if(state==='highlight'){

      for(var i=0;i<E.length;i++) drawEdge(E[i],1,E[i].s?1:0.2);

      for(var i=0;i<E.length;i++) drawBadge(E[i],E[i].s?1:0);

      for(var i=0;i<N.length;i++) drawNode(N[i],1,glow);

      if(tick>280){

        state='appear';tick=0;shown=false;

        for(var i=0;i<N.length;i++) na[i]=0;

        for(var i=0;i<E.length;i++){ep[i]=0;ba[i]=0;}

        if(alertEl)alertEl.classList.remove('visible');

        if(statusDot)statusDot.classList.remove('green');

        if(statusText)statusText.textContent='Scanning incidents';

      }

    }

  }

  frame();

};
